package com.reverse_string;

/* INPUT=Salman Kavthekar
 * OUTPUT=namlaS rakehtvaK
 */

public class ReverseString {
	public static void main(String[] args) {
		
		Scanner scanner=new Scanner(System.in);
		System.out.println("Enter String");   //Salman Kavthekar
		String str= scanner.nextLine();
		
		//[Salman, Kavthekar]
		String[] words=str.split(" ");
		String revString="";
		
		for(int i=0; i<words.length; i++) {
			String word=words[i];
			String revWord="";
			
			//Salman
			for(int j=word.length()-1; j>=0; j--) {
				revWord=revWord+word.charAt(j);  //namlaS
			}
			
			revString=revString+revWord+" ";  //  namlaS rakehtvaK
		}
		
		System.out.println("Reverse Strinf is >> "+revString);
	}
	
	

}
#####OUTPUT=namlaS rakehtvaK

@RestController
@RequestMapping("/api/sort")
public class SortController {

    @GetMapping("/byNameAndCity")
    public List<Person> sortByNameAndCity(@RequestParam(required = false, defaultValue = "asc") String order) {
        List<Person> people = // retrieve the list of people from a database or another source
        if (order.equalsIgnoreCase("asc")) {
            people.sort(Comparator.comparing(Person::getName).thenComparing(Person::getCity));
        } else {
            people.sort(Comparator.comparing(Person::getName).thenComparing(Person::getCity).reversed());
        }
        return people;
    }
}
